        function toggleSidebarMenu() {
            const sidebarMenu = document.getElementById('sidebarMenu');
            
            if (sidebarMenu.classList.contains('show')) {
                // Cerrar menú
                sidebarMenu.classList.remove('show');
                setTimeout(() => {
                    sidebarMenu.style.display = 'none';
                }, 300); // Esperar a que termine la animación
            } else {
                // Abrir menú
                sidebarMenu.style.display = 'flex';
                // Forzar reflow para que la transición funcione
                sidebarMenu.offsetHeight;
                sidebarMenu.classList.add('show');
            }
        }

        function closeSidebarMenu(event) {
            // Solo cerrar si se hace clic en el overlay (fondo), no en el contenido del menú
            if (event.target === event.currentTarget) {
                toggleSidebarMenu();
            }
        }

            // Obtener el grupo de radio del elemento padre

        // Variables para el onboarding
        let currentStep = 0;
        const totalSteps = 7;
        let userResponses = {};

        function nextStep() {
            if (currentStep < totalSteps - 1) {
                // Ocultar paso actual
                const currentStepElement = document.getElementById(`bienvenida-step-${currentStep}`);
                currentStepElement.classList.add('exit');
                currentStepElement.classList.remove('active');
                
                setTimeout(() => {
                    currentStepElement.style.display = 'none';
                    currentStep++;
                    
                    // Mostrar siguiente paso
                    const nextStepElement = document.getElementById(`bienvenida-step-${currentStep}`);
                    nextStepElement.style.display = 'flex';
                    setTimeout(() => {
                        nextStepElement.classList.add('active');
                        nextStepElement.classList.remove('exit');
                    }, 50);
                }, 300);
            }
        }

        function previousStep() {
            if (currentStep > 1) {
                // Ocultar paso actual
                const currentStepElement = document.getElementById(`bienvenida-step-${currentStep}`);
                currentStepElement.classList.add('exit');
                currentStepElement.classList.remove('active');
                
                setTimeout(() => {
                    currentStepElement.style.display = 'none';
                    currentStep--;
                    
                    // Mostrar paso anterior
                    const prevStepElement = document.getElementById(`bienvenida-step-${currentStep}`);
                    prevStepElement.style.display = 'flex';
                    setTimeout(() => {
                        prevStepElement.classList.add('active');
                        prevStepElement.classList.remove('exit');
                    }, 50);
                }, 300);
            }
        }

        function updateActiveOption(selectedBot) {
            // Remover clase active de todas las opciones
            const options = document.querySelectorAll('.dropdown-option');
            options.forEach(option => {
                option.classList.remove('active');
                const badge = option.querySelector('.current-badge');
                if (badge) badge.remove();

            // Agregar clase active a la opción seleccionada
            options.forEach(option => {
                const text = option.querySelector('span:not(.option-icon):not(.current-badge)').textContent;

                    const badge = document.createElement('span');

                    badge.textContent = '(actual)';
                    option.appendChild(badge);
                }
            });

        function sendMessage() {
            const input = document.getElementById('messageInput');
            const messagesContainer = document.getElementById('chatMessages');
            
            if (input.value.trim()) {
                // Crear mensaje del usuario
                const userMessage = document.createElement('div');
                userMessage.className = 'message-wrapper user-message';
                userMessage.innerHTML = `
                    <div class="message user">
                        <div class="message-content">${input.value}</div>
                        <div class="message-time">${getCurrentTime()}</div>
                    </div>
                `;
                
                messagesContainer.appendChild(userMessage);
                
                // Limpiar input
                input.value = '';
                
                // Scroll al final
                messagesContainer.scrollTop = messagesContainer.scrollHeight;
                
                // Simular respuesta de IA después de 1.5 segundos
                setTimeout(() => {
                    const aiMessage = document.createElement('div');
                    aiMessage.className = 'message-wrapper ai-message';
                    aiMessage.innerHTML = `
                        <div class="message ai">
                            <div class="message-content">Perfecto, te ayudo con esa información. ¿Podrías ser más específico sobre qué detalles necesitas?</div>
                            <div class="message-time">${getCurrentTime()}</div>
                        </div>
                    `;
                    
                    messagesContainer.appendChild(aiMessage);
                    messagesContainer.scrollTop = messagesContainer.scrollHeight;
                }, 1500);
            }
        }

        function getCurrentTime() {
            const now = new Date();
            const hours = now.getHours().toString().padStart(2, '0');
            const minutes = now.getMinutes().toString().padStart(2, '0');
            return `${hours}:${minutes}`;
        }

        // Inicializar contador de caracteres
        function initializeCharCounter() {
            const textarea = document.getElementById('promptLibreTextarea');
            const counter = document.getElementById('charCounter');
            
            if (textarea && counter) {
                // Función para actualizar el contador
                function updateCounter() {
                    const currentLength = textarea.value.length;
                    const maxLength = 10000;
                    
                    // Actualizar el texto del contador
                    counter.textContent = `${currentLength}/${maxLength}`;
                    
                    // Cambiar color si excede el límite
                    if (currentLength > maxLength) {
                        counter.classList.add('exceeded');
                    } else {
                        counter.classList.remove('exceeded');
                    }
                }
                
                // Inicializar contador con el contenido actual
                updateCounter();
                
                // Actualizar en tiempo real mientras escribe
                textarea.addEventListener('input', updateCounter);
                textarea.addEventListener('keyup', updateCounter);
                textarea.addEventListener('paste', () => {
                    // Pequeño delay para capturar el contenido después del paste
                    setTimeout(updateCounter, 10);
                });
            }
        }
